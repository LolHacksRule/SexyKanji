virtual int Sexy::SexyFont::GetLineSpacingOffset()
{
    KTrueText* pKTrueText = dynamic_cast<KTrueText*>(mKTextFace);
    if (pKTrueText)
        return (int)K_MAX(mKTextFace->getYSpacing(), mLineSpacingOffset);

    return mLineSpacingOffset;
}
