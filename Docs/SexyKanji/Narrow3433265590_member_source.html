std::string Sexy::StringUtility::Narrow(const std::wstring &str)
{
    std::ostringstream stm;
    const std::ctype<wchar_t>& ctfacet = std::use_facet<std::ctype<wchar_t>>(stm.getloc());

    for (size_t i = 0; i < str.size(); i++)
