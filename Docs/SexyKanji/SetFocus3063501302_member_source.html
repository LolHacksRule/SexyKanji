void Sexy::WidgetManager::SetFocus(Widget *aWidget)
{

        return;

    if (mFocusWidget != NULL)
        mFocusWidget->LostFocus();

    if ((aWidget != NULL) && (aWidget->mWidgetManager == this))
    {
        mFocusWidget = aWidget;
