void Sexy::SexyFPSTimer::OnEvent()
{
    if (mLastCheck == 0)
    {
        mLastCheck = gSexyAppBase->GetClock();
        mTickCount = 0;
        return;
    }

    if(gSexyAppBase->GetClock() - mLastCheck > 1000)
    {
        double aDiff = (float)gSexyAppBase->GetClock() - (float)mLastCheck;
        aDiff = 1000.0 / aDiff;
        mFPS = (long)((float)mTickCount * aDiff);
        mTickCount = 0;
        mLastCheck = gSexyAppBase->GetClock();
    }

    mTickCount++;
}
