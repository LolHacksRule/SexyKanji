void Sexy::XMLParser::SetStringSource(const std::wstring &theString)
{
    Init();

    int aSize = theString.size();

    mBufferedText.resize(aSize); 
    for (int i = 0; i < aSize; i++)
        mBufferedText[i] = theString[aSize - i - 1];
