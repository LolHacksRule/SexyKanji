virtual void Sexy::SexyKanjiTransform::Scale(float sx, float sy)
{
    if (!mComplex)
    {
        if (mHaveRot || mTransX1!=0 || mTransY1!=0 || (sx<0 && mScaleX*sx!=-1) || sy<0)
        {
            MakeComplex();
            
            KMatrix aMat;
            aMat.scaling(sx,sy,1.0f);
            mKMatrix = aMat * mKMatrix;
        }
        else
        {
            mNeedCalcMatrix = true;
            mHaveScale = true;
            mScaleX *= sx;
            mScaleY *= sy;
        }
    }
    else
    {
        KMatrix aMat;
        aMat.scaling(sx,sy,1.0f);
        mKMatrix = aMat * mKMatrix;
    }
}
