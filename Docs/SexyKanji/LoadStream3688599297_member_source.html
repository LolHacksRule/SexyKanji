int Sexy::MusicManager::LoadStream(const std::string &theFileName)
{
    unsigned int aStreamID;
    int aFailedID = -1;

    // if we've already loaded the stream, just return its ID
    for (aStreamID = 0; aStreamID < MAX_MUSIC_STREAMS; aStreamID++)
    {
        if (mStreamsInfo[aStreamID].mFileName == StringToSexyStringFast(theFileName))
        {
            return aStreamID;
        }
    }

    // we haven't already loaded the stream, so find an available ID and 
    // try to load it, then return the new ID
    for (aStreamID = MAX_MUSIC_STREAMS - 1; aStreamID >= 0; aStreamID--)
    {       
        if (mStreamsInfo[aStreamID].mSoundInstance == NULL)
        {
            if (!LoadStream(aStreamID, theFileName))
            {
                return aFailedID;
            }
            else
            {
                return aStreamID;
            }
        }
    }   

    return aFailedID;
}
