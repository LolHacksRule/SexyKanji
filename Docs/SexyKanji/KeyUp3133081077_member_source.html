bool Sexy::WidgetManager::KeyUp(EKeyboardLayout key)
{
    mLastInputUpdateCnt = mUpdateCnt;

    if ((key >= 0) && (key < 0xFF))
        mKeyDown[key] = false;

    if ((key == K_VK_TAB) && (mKeyDown[K_VK_R_CONTROL] || mKeyDown[K_VK_L_CONTROL]))
        return true;    

    if (mFocusWidget != NULL)
        mFocusWidget->KeyUp(key);
    
    return true;
}
