void Sexy::SexyTransform2D::RotateRad(float rot)
{
    SexyMatrix3 aMat;
    aMat.LoadIdentity();

    float sinRot = -sinf(rot);
    float cosRot = cosf(rot);

    aMat.m00 = cosRot;
    aMat.m01 = -sinRot;
    aMat.m10 = sinRot;
    aMat.m11 = cosRot;

    *this = aMat * (*this);
}
