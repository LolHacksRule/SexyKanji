std::wstring Sexy::Buffer::UTF8ToWideString() const
{
    const char* aData = (const char*)GetDataPtr();
    int aLen = GetDataLen();

    bool firstChar = true;

    std::wstring aString;
    aString.reserve(aLen); // worst case
    while (aLen > 0)
    {
        wchar_t aChar;
        int aConsumed = GetUTF8Char(&aData, aLen, &aChar);
        if (aConsumed == 0) break;
        aLen -= aConsumed;

        if (firstChar)
        {
            firstChar = false;
            if (aChar == 0xFEFF) continue;
        }

        aString += aChar;
    }
    return aString;
}
