void Sexy::StringUtility::Tokenize(const std::wstring &theSourceString, std::vector< std::wstring > &theTokens, const std::wstring &theDelimiter=L"")
{
    // If no delimiter is passed, tokenize on all white space.
    if (theDelimiter == L"")
    {
        std::wstring aBuffer; // Have a buffer string
        std::wstringstream ss(theSourceString); // Insert the string into a stream

        while (ss >> aBuffer)
        {
            theTokens.push_back(aBuffer);
        }
    }

    // Skip delimiters at beginning.
    std::wstring::size_type aLastPosition = theSourceString.find_first_not_of(theDelimiter, 0);

    // Find first "non-delimiter".
    std::wstring::size_type aPosition = theSourceString.find_first_of(theDelimiter, aLastPosition);

    while (aPosition != std::wstring::npos || aLastPosition != std::wstring::npos)
    {
        // Found a token, add it to the vector.
        theTokens.push_back(theSourceString.substr(aLastPosition, aPosition - aLastPosition));

        // Skip delimiters.  Note the "not_of"
        aLastPosition = theSourceString.find_first_not_of(theDelimiter, aPosition);

        // Find next "non-delimiter"
        aPosition = theSourceString.find_first_of(theDelimiter, aLastPosition);
    }
}
