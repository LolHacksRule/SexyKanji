virtual void Sexy::Transform::Scale(float sx, float sy)
{
    if (!mComplex)
    {
        if (mHaveRot || mTransX1!=0 || mTransY1!=0 || (sx<0 && mScaleX*sx!=-1) || sy<0)
        {
            MakeComplex();
            mMatrix.Scale(sx,sy);
        }
        else
        {
            mNeedCalcMatrix = true;
            mHaveScale = true;
            mScaleX *= sx;
            mScaleY *= sy;
        }
    }
    else
        mMatrix.Scale(sx,sy);
}
