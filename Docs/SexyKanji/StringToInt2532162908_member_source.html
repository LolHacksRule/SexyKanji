bool StringToInt(const std::wstring theString, int *theIntVal)
{
    *theIntVal = 0;

    if (theString.length() == 0)
        return false;

    int theRadix = 10;
    bool isNeg = false;

    unsigned i = 0;
    if (theString[i] == '-')
    {
        isNeg = true;
        i++;
    }

    for (; i < theString.length(); i++)
    {
        wchar_t aChar = theString[i];
        
        if ((theRadix == 10) && (aChar >= L'0') && (aChar <= L'9'))
            *theIntVal = (*theIntVal * 10) + (aChar - L'0');
        else if ((theRadix == 0x10) && 
            (((aChar >= L'0') && (aChar <= L'9')) || 
             ((aChar >= L'A') && (aChar <= L'F')) || 
             ((aChar >= L'a') && (aChar <= L'f'))))
        {           
            if (aChar <= L'9')

            else if (aChar <= L'F')
                *theIntVal = (*theIntVal * 0x10) + (aChar - L'A') + 0x0A;

        }

        }

        {
            *theIntVal = 0;
            return false;
        }
    }

    if (isNeg)
        *theIntVal = -*theIntVal;

    return true;
