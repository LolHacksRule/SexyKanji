bool Sexy::ResourceManager::Fail(const std::string &theErrorText)
{
    if (mHasFailed == false)
    {
        mHasFailed = true;

        if (mXMLParser == NULL)
        {
            mErrorString = theErrorText;
        }
        else
        {
            // get line info from parser
            int aLineNum = mXMLParser->GetCurrentLineNum();

            std::string aLineNumStr;
            aLineNumStr = StrFormat("%d", aLineNum);

            mErrorString = theErrorText;

            if (aLineNum > 0)
                mErrorString += std::string(" On Line ") + aLineNumStr;

            if (mXMLParser->GetFileName().length() > 0)
                mErrorString += " in File '" + mXMLParser->GetFileName() + "'.";
        }

        // write error to log file, if error is resource manager generated
        if (mErrorString.find("ResourceManager") != std::string::npos)
            K_LOG("Sexy:ResourceManager:Fail: %s", mErrorString.c_str());
    }

    return false;
}
