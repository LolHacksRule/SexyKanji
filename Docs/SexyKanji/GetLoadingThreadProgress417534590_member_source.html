virtual float Sexy::SexyAppBase::GetLoadingThreadProgress()
{
    if (mLoaded)
        return 1.0f;
    if (!mLoadingThreadStarted)
        return 0.0f;
    if (mNumLoadingThreadTasks == 0)
        return 0.0f;
    return K_MIN(mCompletedLoadingThreadTasks / static_cast<float>(mNumLoadingThreadTasks), 1.0f);
}
