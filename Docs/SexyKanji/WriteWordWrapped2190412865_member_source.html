virtual int Sexy::Graphics::WriteWordWrapped(const SexyRect &theRect, const std::wstring &theString, int theLineSpacing=-1, int theJustification=-1, bool doBlit=true)
{
    if (mFont == NULL || theRect.mWidth == 0) return 0;
    PreDraw();

    // If the text won't fit in the rect, we shorten it until it will.
    if ((theRect.mHeight > 0) && (GetWordWrappedHeight(theString, theRect, theLineSpacing, theJustification) > theRect.mHeight))
    {
        std::vector<std::wstring> theTokens;
        StringUtility::Tokenize(theString, theTokens);
        std::wstring aString = theString;

        // Doing an if above and a while here to save the Tokenize step when it is not needed.
        while ((GetWordWrappedHeight(aString, theRect, theLineSpacing, theJustification) > theRect.mHeight) && (!theTokens.empty()))
        {
            aString = aString.substr(0, aString.length() - theTokens.back().length() - 1);
            theTokens.pop_back();
        }
    
        return WriteWordWrapped(aString, theRect.mX, theRect.mY, theRect.mWidth, theLineSpacing, theJustification, doBlit);
    }
    else
    {
        return WriteWordWrapped(theString, theRect.mX, theRect.mY, theRect.mWidth, theLineSpacing, theJustification, doBlit);
    }
}
