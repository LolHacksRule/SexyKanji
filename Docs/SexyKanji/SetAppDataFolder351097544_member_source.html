void SetAppDataFolder(const std::wstring &thePath)
{

        std::wstring aPath = thePath;
        if (!aPath.empty())
        {
            if (aPath[aPath.length()-1] != L'\\' && aPath[aPath.length()-1] != L'/')
                aPath += L'\\';
        }

        Sexy::gAppDataFolder = aPath;
    }
