bool Sexy::XMLWriter::CloseFile()
{
    if(mFileClosed) return true;
    mFileClosed = true;

    while(!mSectionStack.empty())
    {
        StopElement();
    }

    Flush();

    if(mFileName != _S(""))
    {
        return Buffer::WriteBufferToFile(mFileName, mFileBuffer);
    }

    Fail(_S("File not Open"));
    return false;
}
