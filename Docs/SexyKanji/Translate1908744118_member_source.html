virtual void Sexy::SexyKanjiTransform::Translate(float tx, float ty)
{
    if (!mComplex)
    {
        mNeedCalcMatrix = true;
        if (mHaveRot || mHaveScale)
        {
            mTransX2 += tx;
            mTransY2 += ty;
        }
        else
        {
            mTransX1 += tx;
            mTransY1 += ty;
        }
    }
    else
    {
        KMatrix aMat;
        aMat.translation(-tx,ty,0);
        mKMatrix = aMat * mKMatrix;
    }
}
