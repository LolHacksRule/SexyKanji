bool StringToDouble(const std::string theString, double *theDoubleVal)
{
    *theDoubleVal = 0.0;

    if (theString.length() == 0)
        return false;

    bool isNeg = false;

    unsigned i = 0;
    if (theString[i] == '-')
    {
        isNeg = true;
        i++;
    }

    for (; i < theString.length(); i++)
    {
        char aChar = theString[i];

        if ((aChar >= '0') && (aChar <= '9'))
            *theDoubleVal = (*theDoubleVal * 10) + (aChar - '0');
        else if (aChar == '.')
        {
            i++;
            break;
        }
        else
        {
            *theDoubleVal = 0.0;
            return false;
        }
    }

    double aMult = 0.1;
    for (; i < theString.length(); i++)
    {
        char aChar = theString[i];

        if ((aChar >= '0') && (aChar <= '9'))
        {
            *theDoubleVal += (aChar - '0') * aMult; 
            aMult /= 10.0;
        }
        else
        {
            *theDoubleVal = 0.0;
            return false;
        }
    }

    if (isNeg)
        *theDoubleVal = -*theDoubleVal;

    return true;
}
