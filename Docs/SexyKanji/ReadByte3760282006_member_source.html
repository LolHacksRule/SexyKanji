unsigned char Sexy::Buffer::ReadByte() const
{
    if ((mReadBitPos + 7)/8 >= (int)mData.size())
    {       
        return 0; // Underflow
    }

    if (mReadBitPos % 8 == 0)
    {
        unsigned char b = mData[mReadBitPos/8];
        mReadBitPos += 8;
        return b;
    }
    else
    {
        int anOfs = mReadBitPos % 8;
            
        unsigned char b = 0;
        
        b = mData[mReadBitPos/8] >> anOfs;
        b |= mData[(mReadBitPos/8)+1] << (8 - anOfs);
        
        mReadBitPos += 8;      
        
        return b;
    }
}
